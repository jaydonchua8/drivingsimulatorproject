What the finalized prototype does

A joystick controls a motor that provides steering feedback by rotating a wheel mount clockwise/counterclockwise. The goal is to make the “turning sensation” feel responsive and consistent while keeping the system stable and safe.

Current Arduino-style control code (example)
// Example Arduino-style code (placeholder)
// Joystick -> direction + speed -> motor driver pins

const int JOY_X = A0;
const int IN1 = 8;   // H-bridge IN1
const int IN2 = 9;   // H-bridge IN2
const int ENA = 10;  // PWM enable

int deadzone = 60;   // tune to reduce jitter

void setup() {
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENA, OUTPUT);
  Serial.begin(9600);
}

void drive(int dir, int pwm) {
  pwm = constrain(pwm, 0, 255);
  if (dir > 0) { digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); }
  else if (dir < 0) { digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH); }
  else { digitalWrite(IN1, LOW); digitalWrite(IN2, LOW); pwm = 0; }
  analogWrite(ENA, pwm);
}

void loop() {
  int x = analogRead(JOY_X);          // 0..1023
  int centered = x - 512;             // -512..+511
  int mag = abs(centered);

  int dir = 0;
  if (mag > deadzone) dir = (centered > 0) ? 1 : -1;

  int pwm = map(constrain(mag, deadzone, 512), deadzone, 512, 80, 255);
  drive(dir, (dir == 0) ? 0 : pwm);

  Serial.print("x="); Serial.print(x);
  Serial.print(" dir="); Serial.print(dir);
  Serial.print(" pwm="); Serial.println((dir==0)?0:pwm);

  delay(12);
}
